# Étape 1 : Build dans une image légère
FROM node:18-alpine AS build

WORKDIR /app

# Installation des dépendances
COPY package*.json ./
RUN npm ci --omit=dev

# Copie du code source
COPY . .

# Étape 2 : Exécution dans une image sécurisée
FROM node:18-alpine

# Ajout des outils nécessaires pour créer un utilisateur non-root
RUN apk add --no-cache shadow

# Création d’un utilisateur système
RUN addgroup -S nonroot && adduser -S -G nonroot nonroot

WORKDIR /app

# Copie des fichiers depuis l’étape build
COPY --from=build /app /app

# Attribution des droits
RUN chown -R nonroot:nonroot /app

# Sécurité : permissions strictes sur les fichiers sensibles
RUN chmod 600 /app/.env || true

# Variables d’environnement recommandées
ENV NODE_ENV=production
ENV N8N_ENFORCE_SETTINGS_FILE_PERMISSIONS=true

# Port exposé
EXPOSE 4000

# Utilisateur non-root
USER nonroot

# Commande de démarrage
CMD ["node", "server.js"]
